Basic carousel

HTML: 
<div class="basic_carousel_wrapper">

  <div id="basic_carousel">
      <img src="images/panorama.jpg" alt="2" style="float: left"/>
      <p>Lorem ipsum dolor sit amet</p>
  </div>
</div>

<div class="basic_carousel_nvigator">

  <div class="button button-prev" onclick="basic_carousel.skip('prev');"> < prev </div>

  <div class="button button-next" onclick="basic_carousel.skip('next');"> next > </div>

  <div style="clear:both;">

</div>
JS:
var basic_carousel = new $.gcarousel('#basic_carousel',{
   transition_duration: 200,
   shift: 200
});
